#reading a character and displaying it
DATA SEGMENT
MSG1 DB 0AH,0DH,"Enter the character: $"
MSG2 DB 0AH,0DH,"Charater entered is: $"
CHR DB 10 DUP(?)
DATA ENDS
CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:MOV AX,DATA
      MOV DS,AX
      LEA SI,CHR
      MOV AH,09H
      LEA DX,MSG1
      INT 21H
      MOV AH,01H
      INT 21H
      MOV [SI],AL
      MOV AH,09H
      LEA DX,MSG2
      INT 21H
      LEA SI,CHR
      MOV AH,02H
      MOV DL,[SI]
      INT 21H
      MOV AH,4CH
      INT 21H
CODE ENDS
END START
