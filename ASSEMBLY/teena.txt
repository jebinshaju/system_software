
DATA SEGMENT
MSG1 DB 0AH,0DH,"Enter a number: $"
MSG2 DB 0AH,0DH,"Enter number to be searched for: $"
MSG3 DB 0AH,0DH,"Number found$"
MSG4 DB 0AH,0DH,"Number not found$"
A DW 05H DUP(?)
B DW 01H DUP(?)
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START: MOV AX,DATA
       MOV DS,AX
       
       LEA SI,A
       MOV BL,05H
    L1:
       MOV AH,09H
       LEA DX,MSG1
       INT 21H
       INC SI
       CALL P1
       MOV [SI],AL
       DEC SI
       CALL P1
       MOV [SI],AL
       INC SI
       INC SI
       DEC BL
       JNZ L1
       
       MOV AH,09H
       LEA DX,MSG2
       INT 21H
       LEA SI,B
       INC SI
       CALL P1
       MOV [SI],AL
       DEC SI
       CALL P1
       MOV [SI],AL
       
       LEA SI,A
       MOV BX,B
     
       MOV CL,05H
    L2:
       MOV AX,[SI]
       CMP AX,BX
       JZ L3
       INC SI 
       INC SI
       DEC CL
       JNZ L2
       MOV AH,09H
       LEA DX,MSG4
       INT 21H
       
       MOV AH,4CH
       INT 21H
       
    L3:
       MOV AH,09H
       LEA DX,MSG3
       INT 21H
       
       MOV AH,4CH
       INT 21H
       
       P1 PROC
       MOV AH,01H
       INT 21H
       SUB AL,30H
       CMP AL,09H
       JLE L4
       SUB AL,07H
   
   L4: MOV CL,04H
       ROL AL,CL
       MOV CH,AL
       MOV AH,01H
       INT 21H
       SUB AL,30H
       CMP AL,09H
       JLE L5
       SUB AL,07H
     
   L5: ADD AL,CH
       RET 
       P1 ENDP 
       
CODE ENDS
END START
