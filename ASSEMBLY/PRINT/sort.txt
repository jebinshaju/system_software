
DATA SEGMENT
MSG1 DB 0AH,0DH,"Enter the string: $"
MSG2 DB 0AH,0DH,"The sorted string: $"
CHR DB 10D DUP(?)
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA

START: MOV AX,DATA
       MOV DS,AX
       
       MOV AH,09H
       LEA DX,MSG1
       INT 21H
       
       MOV CL,00H	
       LEA SI,CHR	
   
   L1: MOV AH,01H
       INT 21H		
       CMP AL,0DH
       JZ L2
       MOV [SI],AL
       INC SI
       INC CL
       JMP L1
   
   L2: MOV BH,CL
       MOV CH,BH
       
   L5: MOV CL,BH
       LEA SI,CHR
   
   L4: MOV AL,[SI]
       MOV BL,[SI+1]
       CMP AL,BL
       JC L3
       XCHG AL,BL
       MOV [SI],AL
       MOV [SI+1],BL
   L3: INC SI
       DEC CL
       JNZ L4 
       DEC CH
       JNZ L5
       JMP L6
      
   L6: MOV CL,BH
       INC CL
       MOV AH,09H
       LEA DX,MSG2
       INT 21H
       LEA SI,CHR
   L7: MOV AH,02H
       MOV DL,[SI]
       INT 21H
       INC SI
       DEC CL
       JNZ L7
       MOV AH,4CH
       INT 21H
       
CODE ENDS
END START
